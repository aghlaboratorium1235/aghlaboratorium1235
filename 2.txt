#include <cstdlib>
#include <time.h>

using namespace std;

const int LIMIT = 5; // d≥ugoúÊ stosu

// Tutaj deklarujemy typ struktury

struct stack
{
    int w,s[LIMIT];
};

// Funkcja zapisuje na stos

void Zapisz(stack * st, int x)
{
    if(st -> w < LIMIT) st -> s[st -> w++] = x;
    else cout << "STOS PELNY!!!" << endl;
}

// Funkcja odczytuje ze stosu

int Czytaj(stack * st)
{
    if(st -> w) return st -> s[--st -> w];
    cout << "STOS PUSTY!!!" << endl;
    return -1;
}

// Program g≥Ûwny

int main()
{
  stack * stos; // wskaünik
  int i,x;

  stos = new stack;   // tworzymy strukturÍ dynamicznπ

  srand(time(NULL));  // inicjujemy generator pseudolosowy

  stos -> w = 0;      // inicjujemy wskaünik stosu

  cout << "ZAPIS 7 LICZB NA STOS\n";

  for(i = 1; i <= 7; i++)
  {
     x = rand();
     cout << x << endl;
     Zapisz(stos,x);
  }

  cout << "\nODCZYT 7 LICZB ZE STOSU\n";

  for(i = 1; i <= 7; i++)
  {
     x = Czytaj(stos);
     cout << x << endl;
  }

  return 0;
}
